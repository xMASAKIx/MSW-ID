async function startFilteredFetch() { const ppsn = document.getElementById('target_ppsn').value.trim(); const status = document.getElementById('status'); const grid = document.getElementById('item_grid'); const mToken = localStorage.getItem('msw_token'); if (!ppsn || !mToken) return alert("請確認 PPSN 與 msw_token"); status.innerHTML = "⏳ 正在搜尋該玩家的公開商品..."; grid.innerHTML = ""; try { // 改用公開商店搜尋端點，而非 manage 端點 const api = `https://mod-gateway-prd-tokyo-2.nexon.com/mverse/v1/shop/mod/products/search?ppsn=${ppsn}&page=1&size=100`; const res = await fetch(`https://corsproxy.io/?${encodeURIComponent(api)}`, { headers: { "mod-accesstoken": mToken } }); const json = await res.json(); // 公開搜尋的資料結構略有不同，通常在 data.products 中 const products = json.data?.products || []; // 過濾出指定部位 const filtered = products.filter(p => ALLOWED_TYPES.hasOwnProperty(p.avatarType)); if (filtered.length === 0) { status.innerHTML = "⚠️ 該玩家目前沒有公開上架指定類別的商品。"; return; } filtered.forEach(item => { const card = document.createElement('div'); card.className = 'item-card'; card.innerHTML = `
${item.price || 10} wc
📝 商品名稱 ${item.productName}
🆔 商品 ID ${item.items[0]?.itemId || 'N/A'}
🖼️ 圖片網址 ${item.thumbnailUrl}
📦 部位 ${ALLOWED_TYPES[item.avatarType]}
`; grid.appendChild(card); }); status.innerHTML = `✅ 成功獲取玩家 ${ppsn} 的 ${filtered.length} 件公開商品。`; } catch (err) { status.innerHTML = "❌ 請求被攔截或 Token 失效"; } }
