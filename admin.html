window.onload = function() { const savedToken = localStorage.getItem('msw_github_token'); if (savedToken) { document.getElementById('token').value = savedToken; updateStatus("✅ 已讀取儲存的 Token", "gray"); } // --- 自動監控：當你切換回這個分頁時自動執行 --- window.addEventListener('focus', () => { console.log("偵測到頁面聚焦，準備自動讀取剪貼簿..."); autoPasteAndParse(true); // 傳入 true 代表是自動執行的，不彈出錯誤提示 }); }; function updateStatus(msg, color) { const s = document.getElementById('status'); s.innerText = msg; s.style.color = color; } async function autoPasteAndParse(isAuto = false) { try { // 讀取剪貼簿 const text = await navigator.clipboard.readText(); if (!text) return; // 1. 解析 ID (8-10位大寫英數) const idMatch = text.match(/[A-Z0-9]{8,10}/); // 2. 解析價格 const priceMatch = text.match(/(\d+)\s*WC/i); // 3. 解析圖片連結 const imgMatch = text.match(/https:\/\/mod-file\.dn\.nexoncdn\.co\.kr\/[^\s]+/); // 4. 解析名稱 let name = ""; if (text.includes("商品名稱")) { name = text.split("商品名稱")[1].split("\n")[0].trim(); } else { const firstLine = text.split("\n")[0].trim(); if (firstLine.length < 20) name = firstLine; } // 檢查是否為有效的 MSW 資料 (至少要有 ID 或 網址) if (!idMatch && !imgMatch) return; // 填入資料 if (idMatch) document.getElementById('id').value = idMatch[0]; if (priceMatch) document.getElementById('wc').value = priceMatch[1] + "wc"; if (imgMatch) document.getElementById('img').value = imgMatch[0]; if (name) document.getElementById('keyword').value = name; updateStatus("✨ 偵測到剪貼簿，已自動填入！", "#5865F2"); } catch (err) { // 自動執行時如果不成功（例如沒權限），不打擾使用者 if (!isAuto) alert("無法讀取剪貼簿，請確保已授權權限。"); } }
