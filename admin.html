async function startAutoFetch() { const ppsn = document.getElementById('target_ppsn').value.trim(); const status = document.getElementById('status'); const grid = document.getElementById('item_grid'); // 從 localStorage 讀取 Token，確保這是在之前的登入流程中存入的 const token = localStorage.getItem('msw_token'); const uid = localStorage.getItem('msw_uid'); if (!ppsn || !token) { status.innerHTML = "❌ 錯誤：請確保已填寫 PPSN 並檢查 Token 是否已儲存。"; return; } status.innerHTML = "⏳ 正在嘗試獲取作者與裝備資訊..."; grid.innerHTML = ""; try { // 1. 二次查詢：嘗試獲取作者暱稱 const profileApi = `https://mod-gateway-prd-tokyo-2.nexon.com/mod/v1/friends/search/ppsn?ppsn=${ppsn}`; const profileRes = await fetch(`https://corsproxy.io/?${encodeURIComponent(profileApi)}`, { headers: { "mod-accesstoken": token, "mod-user-id": uid } }); const profileJson = await profileRes.json(); // 根據日誌結構解析：data -> result -> [0] -> profileName let authorName = "未知作者"; if (profileJson.data && profileJson.data.result && profileJson.data.result.length > 0) { authorName = profileJson.data.result[0].profileName; } else { // 如果 API 沒回傳名字，至少顯示 PPSN 的後五碼作為識別 authorName = `User_${ppsn.slice(-5)}`; } // 2. 獲取裝備清單 const equipApi = `https://mod-gateway-prd-tokyo-2.nexon.com/mverse/v1/shop/mod/inventory/avatars/manage/equip/list/${ppsn}`; const equipRes = await fetch(`https://corsproxy.io/?${encodeURIComponent(equipApi)}`, { headers: { "mod-accesstoken": token, "mod-user-id": uid } }); const equipJson = await equipRes.json(); const items = equipJson.data?.items || []; if (items.length === 0) { status.innerHTML = `⚠️ [${authorName}] 目前沒有穿戴任何裝備或資料不公開。`; return; } // 3. 渲染卡片 (保持 Tab 縮排與圖片樣式) items.forEach(item => { const card = document.createElement('div'); card.className = 'item-card'; card.innerHTML = `
📝 商品名稱 ${item.itemName}
10 wc
👤 作者 ${authorName}
🆔 商品 ID ${item.itemId}
📦 部位 ${item.avatarType || 'EQUIP'}
`; grid.appendChild(card); }); status.innerHTML = `✅ 成功載入作者 [${authorName}] 的 ${items.length} 個裝備`; } catch (err) { console.error("Fetch Error:", err); status.innerHTML = `❌ 查詢失敗：可能是 Token 過期或網路錯誤。`; } }
